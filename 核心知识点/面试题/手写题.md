### 冒泡排序

```js
let arr = [3, 2, 1, 7,5,7,9,8, 6]
let len = arr.length
for(let i = 0; i < len; i++) {
  for(let j = i + 1; j < len; j ++) {
    if(arr[i] > arr[j]) {
      let temp = arr[i]
      arr[i] = arr[j]
      arr[j] = temp
    }
  }
}
```
